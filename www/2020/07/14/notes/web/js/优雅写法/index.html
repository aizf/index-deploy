9 连接数组

const arr1 = [1,2,3];
const arr2 = [5,6];
// arr1结果
// [1, 2, 3, 5, 6]

// 1, 改变原数组
arr1.push.apply(arr1, arr2);

// 2, 改变原数组
arr1.concat(arr2)

// 3, 不改变原数组
[...arr1, ...arr2]
